```c++
// 设置某一位为1，而保证其它的位不变
result |= (1<<12);		
// 设置某一位为0，而保证其它的位不变
result &= ~(1<<12);
// 获得最低2位
result = exp & 3;
// 将某一位取反，而保证其它的位不变
result ^= (1<<12);
// 将自己清0
result ^= reslut;
```



```c++
// 最右边的1变成0
x &= (x-1);
// 检测一个无符号数是不为2的次方
(x & (x-1) == 0) ? true : false;
```



```c++
// 计算绝对值
int abs(int x)
{
    int y;
    y = x >> 31;
    return (x^y)-y; // or:(x+y)^y
}
```
